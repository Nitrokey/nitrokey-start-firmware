00098 static void serve_interrupt(void) {
00099   uint32_t isr, rsr, tsr;
00100 
00101   /* Fix for the EMAC errata */
00102   isr = AT91C_BASE_EMAC->EMAC_ISR;
00103   rsr = AT91C_BASE_EMAC->EMAC_RSR;
00104   tsr = AT91C_BASE_EMAC->EMAC_TSR;
00105 
00106   if ((isr & AT91C_EMAC_RCOMP) || (rsr & RSR_BITS)) {
00107     if (rsr & AT91C_EMAC_REC) {
00108       chSysLockFromIsr();
00109       chSemResetI(&ETH1.md_rdsem, 0);
00110 #if CH_USE_EVENTS
00111       chEvtBroadcastI(&ETH1.md_rdevent);
00112 #endif
00113       chSysUnlockFromIsr();
00114     }
00115     AT91C_BASE_EMAC->EMAC_RSR = RSR_BITS;
00116   }
00117 
00118   if ((isr & AT91C_EMAC_TCOMP) || (tsr & TSR_BITS)) {
00119     if (tsr & AT91C_EMAC_COMP) {
00120       chSysLockFromIsr();
00121       chSemResetI(&ETH1.md_tdsem, 0);
00122       chSysUnlockFromIsr();
00123     }
00124     AT91C_BASE_EMAC->EMAC_TSR = TSR_BITS;
00125   }
00126   AT91C_BASE_AIC->AIC_EOICR = 0;
00127 }

00165 void mac_lld_init(void) {
00166   unsigned i;
00167 
00168   miiInit();

00170 
00171   /*
00172    * Buffers initialization.
00173    */
00174   for (i = 0; i < EMAC_RECEIVE_DESCRIPTORS; i++) {
00175     rd[i].w1 = (uint32_t)&rb[i * EMAC_RECEIVE_BUFFERS_SIZE];
00176     rd[i].w2 = 0;
00177   }
00178   rd[EMAC_RECEIVE_DESCRIPTORS - 1].w1 |= W1_R_WRAP;
00179   rxptr = rd;
00180   for (i = 0; i < EMAC_TRANSMIT_DESCRIPTORS; i++) {
00181     td[i].w1 = (uint32_t)&tb[i * EMAC_TRANSMIT_BUFFERS_SIZE];
00182     td[i].w2 = EMAC_TRANSMIT_BUFFERS_SIZE | W2_T_LAST_BUFFER | W2_T_USED;
00183   }
00184   td[EMAC_TRANSMIT_DESCRIPTORS - 1].w2 |= W2_T_WRAP;
00185   txptr = td;
00186 
00187   /*
00188    * Associated PHY initialization.
00189    */
00190   miiReset(&ETH1);
00191 
00192   /*
00193    * EMAC pins setup and clock enable. Note, PB18 is not included because it is
00194    * used as #PD control and not as EF100.
00195    */
00196   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_EMAC;
00197   AT91C_BASE_PIOB->PIO_ASR = EMAC_PIN_MASK;
00198   AT91C_BASE_PIOB->PIO_PDR = EMAC_PIN_MASK;
00199   AT91C_BASE_PIOB->PIO_PPUDR = EMAC_PIN_MASK;
00200 
00201   /*
00202    * EMAC Initial setup.
00203    */
00204   AT91C_BASE_EMAC->EMAC_NCR = 0;                /* Stopped but MCE active.*/
00205   AT91C_BASE_EMAC->EMAC_NCFGR = 2 << 10;        /* MDC-CLK = MCK / 32 */
00206   AT91C_BASE_EMAC->EMAC_USRIO = AT91C_EMAC_CLKEN;/* Enable EMAC in MII mode.*/
00207   AT91C_BASE_EMAC->EMAC_RBQP = (AT91_REG)rd;    /* RX descriptors list.*/
00208   AT91C_BASE_EMAC->EMAC_TBQP = (AT91_REG)td;    /* TX descriptors list.*/
00209   AT91C_BASE_EMAC->EMAC_RSR = AT91C_EMAC_OVR |
00210                               AT91C_EMAC_REC |
00211                               AT91C_EMAC_BNA;   /* Clears RSR.*/
00212   AT91C_BASE_EMAC->EMAC_NCFGR |= AT91C_EMAC_DRFCS;/* Initial NCFGR settings.*/
00213   AT91C_BASE_EMAC->EMAC_NCR |= AT91C_EMAC_TE |
00214                                AT91C_EMAC_RE |
00215                                AT91C_EMAC_CLRSTAT;/* Initial NCR settings.*/
00216   mac_lld_set_address(&ETH1, default_mac);
00217 
00218   /*
00219    * PHY device identification.
00220    */
00221   AT91C_BASE_EMAC->EMAC_NCR |= AT91C_EMAC_MPE;
00222   if ((miiGet(&ETH1, MII_PHYSID1) != (PHY_ID >> 16)) ||
00223       ((miiGet(&ETH1, MII_PHYSID2) & 0xFFF0) != (PHY_ID & 0xFFF0)))
00224     chSysHalt();
00225   AT91C_BASE_EMAC->EMAC_NCR &= ~AT91C_EMAC_MPE;
00226 
00227   /*
00228    * Interrupt configuration.
00229    */
00230   AT91C_BASE_EMAC->EMAC_IER = AT91C_EMAC_RCOMP | AT91C_EMAC_TCOMP;

00254 /**
00255  * @brief Returns a transmission descriptor.
00256  * @details One of the available transmission descriptors is locked and
00257  *          returned.
00258  *
00259  * @param[in] macp pointer to the @p MACDriver object
00260  * @param[out] tdp pointer to a @p MACTransmitDescriptor structure
00261  * @return The operation status.
00262  * @retval RDY_OK the descriptor was obtained.
00263  * @retval RDY_TIMEOUT descriptor not available.
00264  */
00265 msg_t max_lld_get_transmit_descriptor(MACDriver *macp,
00266                                       MACTransmitDescriptor *tdp) {
00267   EMACDescriptor *edp;
00268 
00269   (void)macp;
00270 
00271   if (!link_up)
00272     return RDY_TIMEOUT;
00273 
00274   chSysLock();
00275   edp = txptr;
00276   if (!(edp->w2 & W2_T_USED) || (edp->w2 & W2_T_LOCKED)) {
00277     chSysUnlock();
00278     return RDY_TIMEOUT;
00279   }
00280   /*
00281    * Set the buffer size and configuration, the buffer is also marked
00282    * as locked.
00283    */
00284   if (++txptr >= &td[EMAC_TRANSMIT_DESCRIPTORS]) {
00285     edp->w2 = W2_T_LOCKED | W2_T_USED | W2_T_LAST_BUFFER | W2_T_WRAP;
00286     txptr = td;
00287   }
00288   else
00289     edp->w2 = W2_T_LOCKED | W2_T_USED | W2_T_LAST_BUFFER;
00290   chSysUnlock();
00291   tdp->td_offset = 0;
00292   tdp->td_size = EMAC_TRANSMIT_BUFFERS_SIZE;
00293   tdp->td_physdesc = edp;
00294   return RDY_OK;
00295 }

00338 /**
00339  * @brief Returns a receive descriptor.
00340  *
00341  * @param[in] macp pointer to the @p MACDriver object
00342  * @param[out] rdp pointer to a @p MACReceiveDescriptor structure
00343  * @return The operation status.
00344  * @retval RDY_OK the descriptor was obtained.
00345  * @retval RDY_TIMEOUT descriptor not available.
00346  */
00347 msg_t max_lld_get_receive_descriptor(MACDriver *macp,
00348                                      MACReceiveDescriptor *rdp) {
00349   unsigned n;
00350   EMACDescriptor *edp;
00351 
00352   (void)macp;
00353   n = EMAC_RECEIVE_DESCRIPTORS;
00354 
00355   /*
00356    * Skips unused buffers, if any.
00357    */
00358 skip:
00359   while ((n > 0) && !(rxptr->w1 & W1_R_OWNERSHIP)) {
00360     if (++rxptr >= &rd[EMAC_RECEIVE_DESCRIPTORS])
00361       rxptr = rd;
00362     n--;
00363   }
00364 
00365   /*
00366    * Skips fragments, if any, cleaning them up.
00367    */
00368   while ((n > 0) && (rxptr->w1 & W1_R_OWNERSHIP) &&
00369                     !(rxptr->w2 & W2_R_FRAME_START)) {
00370     rxptr->w1 &= ~W1_R_OWNERSHIP;
00371     if (++rxptr >= &rd[EMAC_RECEIVE_DESCRIPTORS])
00372       rxptr = rd;
00373     n--;
00374   }
00375 
00376   /*
00377    * Now compute the total frame size skipping eventual incomplete frames
00378    * or holes...
00379    */
00380 restart:
00381   edp = rxptr;
00382   while (n > 0) {
00383     if (!(rxptr->w1 & W1_R_OWNERSHIP)) {
00384       /* Empty buffer for some reason... cleaning up the incomplete frame.*/
00385       cleanup(edp);
00386       goto skip;
00387     }
00388     /*
00389      * End Of Frame found.
00390      */
00391     if (rxptr->w2 & W2_R_FRAME_END) {
00392       rdp->rd_offset = 0;
00393       rdp->rd_size = rxptr->w2 & W2_T_LENGTH_MASK;
00394       rdp->rd_physdesc = edp;
00395       return RDY_OK;
00396     }
00397 
00398     if ((edp != rxptr) && (rxptr->w2 & W2_R_FRAME_START)) {
00399       /* Found another start... cleaning up the incomplete frame.*/
00400       cleanup(edp);
00401       goto restart;     /* Another start buffer for some reason... */
00402     }
00403 
00404     if (++rxptr >= &rd[EMAC_RECEIVE_DESCRIPTORS])
00405       rxptr = rd;
00406     n--;
00407   }
00408   return RDY_TIMEOUT;
00409 }

00411 /**
00412  * @brief Reads from a receive descriptor's stream.
00413  *
00414  * @param[in] rdp pointer to a @p MACReceiveDescriptor structure
00415  * @param[in] buf pointer to the buffer that will receive the read data
00416  * @param[in] size number of bytes to be read
00417  * @return The number of bytes read from the descriptor's stream, this
00418  *         value can be less than the amount specified in the parameter
00419  *         @p size if there are no more bytes to read.
00420  */
00421 size_t mac_lld_read_receive_descriptor(MACReceiveDescriptor *rdp,
00422                                          uint8_t *buf,
00423                                          size_t size) {
00424   if (size > rdp->rd_size - rdp->rd_offset)
00425     size = rdp->rd_size - rdp->rd_offset;
00426   if (size > 0) {
00427     uint8_t *src = (uint8_t *)(rdp->rd_physdesc->w1 & W1_R_ADDRESS_MASK) +
00428                    rdp->rd_offset;
00429     uint8_t *limit = &rb[EMAC_RECEIVE_DESCRIPTORS * EMAC_RECEIVE_BUFFERS_SIZE];
00430     if (src >= limit)
00431       src -= EMAC_RECEIVE_DESCRIPTORS * EMAC_RECEIVE_BUFFERS_SIZE;
00432     if (src + size > limit ) {
00433       memcpy(buf, src, (size_t)(limit - src));
00434       memcpy(buf + (size_t)(limit - src), rb, size - (size_t)(limit - src));
00435     }
00436     else
00437       memcpy(buf, src, size);
00438     rdp->rd_offset += size;
00439   }
00440   return size;
00441 }

00444  * @brief Releases a receive descriptor.
00445  * @details The descriptor and its buffer are made available for more incoming
00446  *          frames.
00447  *
00448  * @param[in] rdp the pointer to the @p MACReceiveDescriptor structure

00450 void mac_lld_release_receive_descriptor(MACReceiveDescriptor *rdp) {
00451   bool_t done;
00452   EMACDescriptor *edp = rdp->rd_physdesc;
00453 
00454   unsigned n = EMAC_RECEIVE_DESCRIPTORS;
00455   do {
00456     done = ((edp->w2 & W2_R_FRAME_END) != 0);
00457     chDbgAssert(edp->w1 & W1_R_OWNERSHIP,

00467    * Make rxptr point to the descriptor where the next frame will most

00470   rxptr = edp;
