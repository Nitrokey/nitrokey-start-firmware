00182 /**
00183  * @brief   Platform dependent part of the @p chThdInit() API.
00184  * @details This code usually setup the context switching frame represented
00185  *          by an @p intctx structure.
00186  */
00187 #define SETUP_CONTEXT(workspace, wsize, pf, arg) {                          \
00188   uint8_t *sp = (uint8_t *)workspace + wsize - sizeof(struct eabi_frame);   \
00189   ((struct eabi_frame *)sp)->slink = 0;                                     \
00190   ((struct eabi_frame *)sp)->shole = _port_thread_start;                    \
00191   tp->p_ctx.sp = (struct intctx *)(sp - sizeof(struct intctx));             \
00192   tp->p_ctx.sp->r31 = arg;                                                  \
00193   tp->p_ctx.sp->r30 = pf;                                                   \
00194 }
