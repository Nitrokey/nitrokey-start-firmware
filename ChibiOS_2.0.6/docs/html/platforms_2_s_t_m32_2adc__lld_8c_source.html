00061 /**
00062  * @brief ADC1 DMA interrupt handler (channel 1).
00063  */
00064 CH_IRQ_HANDLER(Vector6C) {
00065   uint32_t isr;
00066 
00067   CH_IRQ_PROLOGUE();
00068 
00069   isr = DMA1->ISR;
00070   DMA1->IFCR |= DMA_IFCR_CGIF1  | DMA_IFCR_CTCIF1 |
00071                 DMA_IFCR_CHTIF1 | DMA_IFCR_CTEIF1;
00072   if ((isr & DMA_ISR_HTIF1) != 0) {
00073     /* Half transfer processing.*/
00074     if (ADCD1.ad_callback != NULL) {
00075       /* Invokes the callback passing the 1st half of the buffer.*/
00076       ADCD1.ad_callback(ADCD1.ad_samples, ADCD1.ad_depth / 2);
00077     }
00078   }
00079   if ((isr & DMA_ISR_TCIF1) != 0) {
00080     /* Transfer complete processing.*/
00081     if (!ADCD1.ad_grpp->acg_circular) {
00082       /* End conversion.*/
00083       adc_lld_stop_conversion(&ADCD1);
00084       ADCD1.ad_grpp  = NULL;
00085       ADCD1.ad_state = ADC_COMPLETE;
00086       chSysLockFromIsr();
00087       chSemResetI(&ADCD1.ad_sem, 0);
00088       chSysUnlockFromIsr();
00089     }
00090     /* Callback handling.*/
00091     if (ADCD1.ad_callback != NULL) {
00092       if (ADCD1.ad_depth > 1) {
00093         /* Invokes the callback passing the 2nd half of the buffer.*/
00094         size_t half = ADCD1.ad_depth / 2;
00095         ADCD1.ad_callback(ADCD1.ad_samples + half, half);
00096       }
00097       else {
00098         /* Invokes the callback passing the whole buffer.*/
00099         ADCD1.ad_callback(ADCD1.ad_samples, ADCD1.ad_depth);
00100       }
00101     }
00102   }
00103   if ((isr & DMA_ISR_TEIF1) != 0) {
00104     /* DMA error processing.*/
00105     STM32_ADC1_DMA_ERROR_HOOK();
00106   }
00107 
00108   CH_IRQ_EPILOGUE();
00109 }

00118  */

00120 
00121 #if USE_STM32_ADC1
00122   /* ADC reset, ensures reset state in order to avoid trouble with JTAGs.*/
00123   RCC->APB2RSTR = RCC_APB2RSTR_ADC1RST;
00124   RCC->APB2RSTR = 0;
00125 
00126   /* Driver initialization.*/
00127   adcObjectInit(&ADCD1);
00128   ADCD1.ad_adc = ADC1;
00129   ADCD1.ad_dma = DMA1_Channel1;

00131 
00132   /* Temporary activation.*/
00133   RCC->APB2ENR |= RCC_APB2ENR_ADC1EN;
00134   ADC1->CR1 = 0;
00135   ADC1->CR2 = ADC_CR2_ADON;
00136 
00137   /* Reset calibration just to be safe.*/
00138   ADC1->CR2 = ADC_CR2_ADON | ADC_CR2_RSTCAL;
00139   while ((ADC1->CR2 & ADC_CR2_RSTCAL) != 0)
00140     ;
00141 
00142   /* Calibration.*/
00143   ADC1->CR2 = ADC_CR2_ADON | ADC_CR2_CAL;
00144   while ((ADC1->CR2 & ADC_CR2_CAL) != 0)
00145     ;
00146 
00147   /* Return the ADC in low power mode.*/
00148   ADC1->CR2 = 0;
00149   RCC->APB2ENR &= ~RCC_APB2ENR_ADC1EN;
00150 #endif
00151 }

00158 void adc_lld_start(ADCDriver *adcp) {
00159 
00160   /* If in stopped state then enables the ADC and DMA clocks.*/
00161   if (adcp->ad_state == ADC_STOP) {
00162 #if USE_STM32_ADC1
00163     if (&ADCD1 == adcp) {
00164       dmaEnable(DMA1_ID);   /* NOTE: Must be enabled before the IRQs.*/
00165       NVICEnableVector(DMA1_Channel1_IRQn,
00166                        CORTEX_PRIORITY_MASK(STM32_ADC1_IRQ_PRIORITY));
00167       DMA1_Channel1->CPAR = (uint32_t)&ADC1->DR;
00168       RCC->APB2ENR |= RCC_APB2ENR_ADC1EN;
00169     }
00170 #endif
00171 
00172     /* ADC setup, the calibration procedure has already been performed
00173        during initialization.*/
00174     adcp->ad_adc->CR1 = ADC_CR1_SCAN;
00175     adcp->ad_adc->CR2 = 0;
00176   }

00200 /**
00201  * @brief Starts an ADC conversion.
00202  *
00203  * @param[in] adcp      pointer to the @p ADCDriver object
00204  */
00205 void adc_lld_start_conversion(ADCDriver *adcp) {
00206   uint32_t ccr, n;
00207   const ADCConversionGroup *grpp = adcp->ad_grpp;
00208 
00209   /* DMA setup.*/
00210   adcp->ad_dma->CMAR  = (uint32_t)adcp->ad_samples;
00211   ccr = adcp->ad_dmaprio | DMA_CCR1_EN   | DMA_CCR1_MSIZE_0 | DMA_CCR1_PSIZE_0 |
00212                            DMA_CCR1_MINC | DMA_CCR1_TCIE    | DMA_CCR1_TEIE;
00213   if (grpp->acg_circular)
00214     ccr |= DMA_CCR1_CIRC;
00215   if (adcp->ad_depth > 1) {
00216     /* If the buffer depth is greater than one then the half transfer interrupt
00217        interrupt is enabled in order to allows streaming processing.*/
00218     ccr |= DMA_CCR1_HTIE;
00219     n = (uint32_t)grpp->acg_num_channels * (uint32_t)adcp->ad_depth;
00220   }
00221   else
00222     n = (uint32_t)grpp->acg_num_channels;
00223   adcp->ad_dma->CNDTR = n;
00224   adcp->ad_dma->CCR = ccr;
00225 
00226   /* ADC setup.*/
00227   adcp->ad_adc->SMPR1 = grpp->acg_smpr1;
00228   adcp->ad_adc->SMPR2 = grpp->acg_smpr2;
00229   adcp->ad_adc->SQR1  = grpp->acg_sqr1;
00230   adcp->ad_adc->SQR2  = grpp->acg_sqr2;
00231   adcp->ad_adc->SQR3  = grpp->acg_sqr3;
00232   adcp->ad_adc->CR1   = grpp->acg_cr1 | ADC_CR1_SCAN;
00233   adcp->ad_adc->CR2   = grpp->acg_cr2 | ADC_CR2_DMA | ADC_CR2_ADON;
00234 
00235   /* ADC start.*/
00236   adcp->ad_adc->CR2  |= ADC_CR2_SWSTART | ADC_CR2_EXTTRIG;
00237 }
